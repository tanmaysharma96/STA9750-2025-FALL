# --- helper to standardize "missing code" placeholders ---
fix_missing_code <- function(x) {
  x <- str_squish(as.character(x))
  x <- na_if(x, "#")
  x <- na_if(x, "")
  x <- na_if(x, "Not assigned")
  x
}

df <- read_excel("C:/Users/tanma/Downloads/Consultancy-1.xlsx", sheet = "Sheet1")

df <- df %>%
  mutate(
    po_uac_amount = as.numeric(`PO UAC AMOUNT`),
    workdays      = as.numeric(`ACTUAL NB WKD`),

    # clean placeholder codes
    DEPARTMENT    = fix_missing_code(DEPARTMENT),
    `WBS NAME`    = fix_missing_code(`WBS NAME`),
    WBS           = fix_missing_code(WBS),
    `Fund Centre` = fix_missing_code(`Fund Centre`),
    REGION        = fix_missing_code(REGION)
  ) %>%
  # Exclude travel-related cost records using COST TYPE
  filter(!str_detect(str_to_lower(`COST TYPE`), "travel"))

0.3 üóÇÔ∏è Data Overview (analytic scope)

‚úì Dataset includes awarded consultancy records with contract value (PO UAC AMOUNT) and duration (ACTUAL NB WKD)
‚úì Daily rates are computed only for individual consultants and only when workdays are valid (> 0)
‚úì Firm-level records are retained for total value comparisons, but excluded from daily rate analysis

‚Üí This section clarifies the analytic sample and ensures consistent interpretation of daily rate metrics throughout the report.

0.4 üßπ Data Quality & Cleaning Rules

‚úì Contract value and workday fields are coerced to numeric for calculation consistency
‚úì Records with missing or non-positive workdays are excluded from daily rate calculations
‚úì Extreme values are retained to reflect genuine heterogeneity in consultancy engagements

‚Üí Cleaning rules ensure daily rates are computed reliably and remain defensible for reporting.

Show code 
# Coerce numeric fields
df <- df %>%
  mutate(
    po_uac_amount = as.numeric(`PO UAC AMOUNT`),
    workdays      = as.numeric(`ACTUAL NB WKD`)
  )

# Contract-level daily rate (prevents overweighting split-coded line items)
ind_contract <- df %>%
  filter(Type_Consultant == "INDIVIDUAL") %>%
  filter(!is.na(workdays), workdays > 0) %>%
  group_by(Contract) %>%
  summarise(
    NAME       = first(NAME),
    DEPARTMENT = first(DEPARTMENT),
    REGION     = first(REGION),
    `WBS NAME` = first(`WBS NAME`),
    po_uac_amount = sum(po_uac_amount, na.rm = TRUE),
    workdays      = max(workdays, na.rm = TRUE),
    .groups = "drop"
  ) %>%
  mutate(
    daily_rate = po_uac_amount / workdays,
    NAME_clean = str_to_upper(str_squish(NAME))
  ) %>%
  filter(is.finite(daily_rate), !is.na(daily_rate))

ind <- ind_contract

0.9 üåç Local vs International Consultants (alternative visualizations ‚Äî clearer than boxplots)

‚úì Boxplots can hide density and overlap when distributions are skewed
‚úì Showing distribution shape or central tendency + spread improves interpretability
‚úì Alternatives below are better suited for executive / audit audiences

‚Üí Use one of the following depending on whether you want emphasis on distribution, comparison, or magnitude.

0.10 Option 1: üéª Violin Plot (distribution shape)

‚úì Shows full density of daily rates by group
‚úì Makes skewness and tail behavior explicit
‚úì More informative than boxplots for log-scaled data

‚Üí Best when explaining how rates differ, not just that they differ.

Show code 
ggplot(ind_geo, aes(x = consultant_origin, y = daily_rate)) +
  geom_violin(trim = FALSE, alpha = 0.6) +
  geom_boxplot(width = 0.15, outlier.shape = NA, alpha = 0.5) +
  scale_y_log10(labels = scales::comma) +
  labs(
    title = "Figure. Distribution of Daily Rates by Consultant Origin",
    x = "Consultant Origin",
    y = "Daily Rate (log scale)"
  )

0.11 Option 2: üìä Median + IQR Dot Plot (executive-friendly)

‚úì Focuses attention on typical pricing, not extremes
‚úì Easier to read for non-technical stakeholders
‚úì Avoids visual clutter from outliers

‚Üí Best for slides, executive summaries, and audit appendices.

Show code 
geo_stats <- ind_geo %>%
  group_by(consultant_origin) %>%
  summarise(
    median_rate = median(daily_rate),
    q25 = quantile(daily_rate, 0.25),
    q75 = quantile(daily_rate, 0.75),
    .groups = "drop"
  )

ggplot(geo_stats, aes(x = consultant_origin, y = median_rate)) +
  geom_point(size = 3) +
  geom_errorbar(
    aes(ymin = q25, ymax = q75),
    width = 0.15
  ) +
  scale_y_log10(labels = scales::comma) +
  labs(
    title = "Figure. Median Daily Rates by Consultant Origin (IQR)",
    x = "Consultant Origin",

  )

0.12 Option 3: üìà ECDF Comparison (distribution dominance)

‚úì Shows whether one group consistently dominates another
‚úì No binning, no assumptions about shape
‚úì Extremely robust for skewed data

‚Üí Best for defensibility: clearly shows stochastic dominance.

Show code 
ggplot(ind_geo, aes(x = daily_rate, color = consultant_origin)) +
  stat_ecdf(size = 1) +
  scale_x_log10(labels = scales::comma) +
  labs(
    title = "Figure. ECDF of Daily Rates by Consultant Origin",
    x = "Daily Rate (log scale)",
    y = "Cumulative Share of Contracts",
    color = "Consultant Origin"
  )

0.13 Use Cases for the 3 options Given Above

‚úì Use Violin in technical annexes

‚Üí Together, these views communicate both rigor and clarity without relying on boxplots.
